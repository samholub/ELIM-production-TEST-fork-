function sanitize(key, data, initial) {
  const schema = SCHEMA_TYPES[key];
  if (!schema) return data;
  try {
    if (schema === "array" || schema === "array-string") {
      if (!Array.isArray(data)) { console.warn("sanitize: expected array for", key); return initial; }
      if (schema === "array-string") {
        const cleaned = data.filter(v => typeof v === "string");
        if (cleaned.length !== data.length) console.warn("sanitize: dropped non-string entries from", key);
        return cleaned;
      }
      return data;
    }
    if (schema === "object") {
      if (typeof data !== "object" || data === null || Array.isArray(data)) {
        console.warn("sanitize: expected object for", key);
        return initial;
      }
      return data;
    }
    if (schema.startsWith("record-")) {
      if (typeof data !== "object" || data === null || Array.isArray(data)) {
        console.warn("sanitize: expected record for", key);
        return initial;
      }
      const expectedType = schema.split("-")[1];
      const cleaned = {};
      let dropped = 0;
      for (const k of Object.keys(data)) {
        if (typeof data[k] === expectedType) { cleaned[k] = data[k]; }
        else { dropped++; }
      }
      if (dropped > 0) console.warn("sanitize: dropped", dropped, "invalid fields from", key);
      return cleaned;
    }
  } catch (e) {
    console.warn("sanitize: error validating", key, e);
    return initial;
  }
  return data;
}

  // Future migrations go here as functions: (storageGet, storageSet) => Promise
  // Index 0 = migration from version 0→1, index 1 = migration from 1→2, etc.

async function runMigrations() {
  try {
    let stored = 0;
    try {
      const r = await window.storage.get("elim3-schema-version", true);
      if (r?.value) stored = JSON.parse(r.value);
    } catch (e) {}
    if (typeof stored !== "number" || stored < 0) stored = 0;
    if (stored >= SCHEMA_VERSION) return;
    for (let i = stored; i < SCHEMA_VERSION && i < MIGRATIONS.length; i++) {
      console.log("Running migration", i, "→", i + 1);
      await MIGRATIONS[i](
        (k) => window.storage.get(k, true).then(r => r?.value ? JSON.parse(r.value) : null),
        (k, v) => window.storage.set(k, JSON.stringify(v), true)
      );
    }
    await window.storage.set("elim3-schema-version", JSON.stringify(SCHEMA_VERSION), true);
    console.log("Schema version set to", SCHEMA_VERSION);
  } catch (e) {
    console.warn("Migration error:", e);
  }
}
