javascriptfunction useStorage(key, initial, shared = true) {
  const [val, setVal] = useState(initial);
  const [loaded, setLoaded] = useState(false);
  const prevRef = useRef(initial);
  useEffect(() => {
    window.storage.get(key, shared).then(r => {
      if (r?.value) {
        try {
          const parsed = JSON.parse(r.value);
          if (parsed != null) { setVal(parsed); prevRef.current = parsed; }
        } catch(err) {}
      }
    }).catch(() => {}).finally(() => setLoaded(true));
    // Real-time sync: update state when Firebase pushes changes from other devices
    const handler = (e) => {
      if (e.detail?.key === key && e.detail?.value) {
        try {
          const parsed = JSON.parse(e.detail.value);
          if (parsed != null) { setVal(parsed); prevRef.current = parsed; }
        } catch(err) {}
      }
    };
    window.addEventListener('elim-sync', handler);
    return () => window.removeEventListener('elim-sync', handler);
  }, []);
  const save = useCallback((v) => {
    const prev = prevRef.current;
    prevRef.current = v;
    setVal(v);
    // Smart merge: if both old and new values are plain objects,
    // diff them and write only changed fields via storage.merge.
    // Full replacements (resets to {}, arrays, scalars) use storage.set.
    if (typeof v === 'object' && v !== null && !Array.isArray(v) &&
        typeof prev === 'object' && prev !== null && !Array.isArray(prev)) {
      const fields = {};
      let changeCount = 0;
      for (const k of Object.keys(v)) {
        if (v[k] !== prev[k]) { fields[k] = v[k]; changeCount++; }
      }
      for (const k of Object.keys(prev)) {
        if (!(k in v)) { fields[k] = null; changeCount++; }
      }
      const totalKeys = Math.max(Object.keys(v).length, Object.keys(prev).length);
      if (changeCount > 0 && changeCount < totalKeys) {
        window.storage.merge(key, fields, shared).catch(() => {});
        return;
      }
    }
    window.storage.set(key, JSON.stringify(v), shared).catch(() => {});
  }, [key, shared]);
  return [val, save, loaded];
}

How it works end-to-end
When a volunteer checks a box, toggleCheck calls setChecks({ ...checks, [itemId]: true }). The save callback diffs against prevRef and sees one field changed out of 59. It routes to storage.merge({ [itemId]: true }) instead of storage.set(fullObject).
storage.merge reads the Firebase node. If it's still in the old JSON string format (first call after deploy), it writes the full native object via .set() — migrating the format. On all subsequent calls, the node is a native object, so it uses .update({ [itemId]: true }) — writing only the changed field.
If two devices are offline and check different tasks, Device A's .update({ ps-1: true }) and Device B's .update({ sb-1: true }) both land independently without clobbering each other.
Full-object resets like setChecks({}) still use storage.set because the diff detects all keys were deleted (not a partial change).
Deploy strategy
Deploy normalization (Edits 2A, 2B) and Fix 1 first, verify for one service. Then deploy the full set. The normalization ensures all devices can read both formats before any device starts writing native objects. Since there's no service worker, a hard refresh picks up new code immediately — but if a volunteer has a stale tab, the normalization protects against format mismatch.
