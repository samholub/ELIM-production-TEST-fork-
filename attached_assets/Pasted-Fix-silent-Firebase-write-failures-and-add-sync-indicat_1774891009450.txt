With this improved version that has more retries, exponential backoff, and dispatches a sync-failure event:
js_pendingWrites.add(key);
const attemptWrite = (attemptsLeft, delay) => {
    _fb.child(fbKey).set(value).then(() => {
        _pendingWrites.delete(key);
        window.dispatchEvent(new CustomEvent('elim-sync-status', { detail: { status: 'ok' } }));
    }).catch((e) => {
        console.warn('Firebase write failed:', e.message, '— retries left:', attemptsLeft);
        if (attemptsLeft > 0) {
            setTimeout(() => attemptWrite(attemptsLeft - 1, Math.min(delay * 2, 30000)), delay);
        } else {
            // Keep pending flag — do NOT let listener overwrite local data
            window.dispatchEvent(new CustomEvent('elim-sync-status', { detail: { status: 'error', key } }));
        }
    });
};
attemptWrite(4, 2000);
Key changes: 5 total attempts with exponential backoff (2s → 4s → 8s → 16s → 30s), does NOT clear _pendingWrites on final failure (prevents listener from reverting local data), and dispatches a elim-sync-status event so the UI can react.
